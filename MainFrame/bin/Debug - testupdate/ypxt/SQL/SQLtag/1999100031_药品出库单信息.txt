
select top 1 out_type,dept_sn,ward_sn,drawer,group_no_in
into #out_info
from yp_out_detl
where out_date=:out_date and draw_no=:draw_no

select a.*,outtype.name as out_name,
       dept.name as dept_name,
       groupname.dept_name as group_name,
       drawer.name as drawer_name,
       0 as plan_pageno
from #out_info a left join yp_in_out_type outtype on a.out_type=outtype.code
    left join yp_group_name groupname on a.group_no_in=groupname.group_no
    left join zd_unit_code dept on a.dept_sn=dept.code
    left join a_employee_mi drawer on drawer.code=a.drawer
drop table #out_info
