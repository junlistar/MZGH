declare @group_no varchar(10)
set @group_no=:group_no
select yp_base.charge_code,yp_base.serial,yp_base.stock_amount,
  yp_dict.specification,yp_dict.drugname,yp_dict.d_code 
 ,yp_dict.retprice,yp_dict.fix_price,yp_base.goods_where_code 
 from yp_dict,yp_base 
 where yp_base.charge_code=yp_dict.charge_code 
 and yp_base.serial=yp_dict.serial 
 and yp_base.group_no=@group_no
 and yp_dict.deleted_flag='0'
and yp_base.stock_amount<>0 
 order by isnull(yp_base.goods_where_code,'æ— '), 
      yp_dict.extend_code,
      yp_dict.charge_code,yp_dict.serial
