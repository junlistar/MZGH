select  a.*,b.py_code,b.d_code,b.drugname,b.specification,m.manu_name,
b.extend_code,case out_type when '00' then '平' when '23' then  '盈' when '24' then '亏' end check_flag,
b.drug_kind
from yp_out_detl a left join yp_dict b on a.charge_code =b.charge_code
  and a.serial =b.serial 
  left join yp_manufacture m on a.manufacture=m.manu_code
where group_no_out = :group_no and
      draw_no=:draw_no and
  Datediff(day,out_date,:out_date)=0 
 order by a.out_seri,b.py_code
