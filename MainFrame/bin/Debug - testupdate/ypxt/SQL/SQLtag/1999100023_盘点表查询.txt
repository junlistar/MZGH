/*
盘点表查询
*/
select keeper,draw_no,out_date,group_no_in,accept_flag,
       -sum(retprice*issue_amount) retprice_total,
       -sum(isnull(fix_price,0)*issue_amount) fix_price_total,
       case accept_flag when '1' then '√' else '' end audit
from yp_out_detl
 where group_no_out = :group_no_out
  and out_date >=:begin_date and out_date< :end_date
  and out_type in('00','23','24')
group by draw_no,out_date,group_no_in,keeper,accept_flag
