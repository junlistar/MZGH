declare @group_no_in varchar(10)
declare @begin_date datetime 
declare @end_date datetime,@group_no_out varchar(10)
select @group_no_in=:group_no_in,
       @begin_date=:begin_date,
       @end_date=:end_date,
       @group_no_out=:group_no_out

select page_no,in_date,flag,case flag when '1' then '√' else '' end as flag1
from yp_in_plan
where (group_no_in = @group_no_in)
and in_date >= @begin_date 
and in_date < cast(@end_date as datetime)
and (group_no_out like @group_no_out )
and (flag='1' or flag='0') /*0,1：请领单 ，4,5采购单*/
and plan_type='1'
group by page_no,in_date,flag
